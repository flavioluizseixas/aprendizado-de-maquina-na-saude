# Instale os pacotes, caso ainda no os tenha
if (!require("TH.data")) install.packages("TH.data")
if (!require("survival")) install.packages("survival")
if (!require("survminer")) install.packages("survminer")

# Carregar os pacotes
library(TH.data)
library(survival)
library(survminer)
library(ggplot2)

# Carregar o conjunto de dados
data("GBSG2", package = "TH.data")
head(GBSG2)
# Criando o objeto de sobrevivncia
surv_obj <- Surv(time = GBSG2$time, event = GBSG2$cens)

# Ajuste do modelo Kaplan-Meier
km_fit <- survfit(surv_obj ~ 1)

# Visualizao do Kaplan-Meier
ggsurvplot(km_fit, data = GBSG2, conf.int = TRUE, 
           title = "Curva de Sobrevivncia Kaplan-Meier",
           xlab = "Tempo (dias)", ylab = "Probabilidade de Sobrevivncia")

# Ajuste do modelo Weibull sem decomposio
wbmod <- survreg(Surv(time, cens) ~ 1, data = GBSG2, dist = "weibull")

# Definir uma sequncia de probabilidades de sobrevivncia
surv <- seq(0.99, 0.01, by = -0.01)

# Calcular os tempos preditos para cada nvel de probabilidade de sobrevivncia
# Como estamos prevendo apenas uma curva mdia, precisamos de um nico conjunto de predies
time_pred <- predict(wbmod, type = "quantile", p = 1 - surv, newdata = data.frame(1))

# Criar um data frame para visualizao
surv_wbmod <- data.frame(time = time_pred, survival = surv)

# Plotar a curva de sobrevivncia Weibull simplificada
ggplot(surv_wbmod, aes(x = time, y = survival)) +
  geom_line(color = "blue") +
  labs(title = "Curva de Sobrevivncia Weibull (Sem Decomposio)",
       x = "Tempo (dias)", y = "Probabilidade de Sobrevivncia") +
  theme_minimal()

# Ajuste do modelo de Cox sem especificar covariveis (curva mdia de sobrevivncia)
cox_fit <- coxph(Surv(time, cens) ~ 1, data = GBSG2)

# Gerar a curva de sobrevivncia mdia
cox_surv <- survfit(cox_fit)

# Visualizar a curva de sobrevivncia usando ggsurvplot
ggsurvplot(cox_surv, data = GBSG2, 
           title = "Curva de Sobrevivncia do Modelo de Cox",
           xlab = "Tempo (dias)", ylab = "Probabilidade de Sobrevivncia",
           conf.int = TRUE)
